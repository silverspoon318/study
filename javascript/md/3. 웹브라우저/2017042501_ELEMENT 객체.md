jQuery객체에는 조회된 객체의 정보가 담겨 있다. 가져온 객체는 jQuery객체가 아니라 DOM객체이다.

사용할려면 jQuery 함수를 사용해야 한다.
$('li').css('', '');

DOM객체로 jQuery함수를 사용하면 jQuery객체가 만들어진다.

## .map()
var li = $('li');
li.map(function(index, elem){
    console.log(index, elem); //elem 은 HTMLLielement 즉 DOM 객체이다.
    $(elem).css('color', 'red');
  });
